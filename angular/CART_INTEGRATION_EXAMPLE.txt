/**
 * Interface pour l'intégration du panier avec le système de stock
 * 
 * Si vous voulez intégrer le système de panier avec la gestion de stock,
 * ajoutez cette interface dans vos modèles et utilisez-la dans vos services.
 */

// 1. Créer l'interface CartItem
export interface CartItem {
  product: Product;
  quantity: number;
}

// 2. Créer l'interface pour les items de transaction
export interface TransactionItem {
  sku: string;
  quantity: number;
  price: number;
  productName?: string;
}

// 3. Étendre l'interface Transaction existante dans api.service.ts
export interface Transaction {
  id?: number;
  userId?: number;
  amount: number;
  paymentMethod: string;
  deviceType?: string;
  ipAddress?: string;
  locationCountry?: string;
  merchantName?: string;
  transactionType?: string;
  status?: string;
  fraudScore?: number;
  riskLevel?: string;
  isFraud?: boolean;
  fraudReasons?: string;
  transactionDate?: string;
  createdAt?: string;
  items?: TransactionItem[];  // ⚠️ AJOUTER CETTE LIGNE
}

/**
 * Note importante sur la décrémentation du stock:

 * manuellement dans le frontend.
